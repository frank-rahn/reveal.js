            package de.rahn.kotlin.lang

            enum class Status {
              OK, ERROR
            }

            fun controlFlowWhen(y: Any) {
              val b = when (y) {
                is String -> true
                else -> false
              }

              val x = 42
              when (x) {
                1 -> println("x ist 1")
                2 -> println("x ist 2")
                3, 4 -> println("x ist 3 oder 4")
                in 5..10 -> println("x ist zwischen 5 und 10")
                else -> println("x ist irgend etwas anderes")
              }

              val status = Status.OK
              when (status) {
                Status.OK -> println("Super")
                Status.ERROR -> println("Nicht gut")
                else -> return
              }

              when {
                x < 42 -> println("-1")
                x == 0 -> println("0")
                x > 42 -> println("1")
                else -> throw IllegalArgumentException("Mist")
              }

              when (val z = max(4, 5)) {
                in 5..100 -> println(z - 5)
              }
            }

            fun main() = controlFlowWhen("")
* Die Kotlin Klasse Any entspricht der Klasse Object von Java.
* Mehrere Bedingungen (Conditions) werden mit einem Komma getrennt
* Es muß kein Ausdruck übergeben werden
* Statt max() sollte natürlich eine Member-Funktion verwendet werden

Tail-recursive

Verwendet Tail-recursion/call oder Endrekursion

Datei: FunctionsTailRecursion.kt

            package de.rahn.kotlin.lang

            import kotlin.math.abs

            const val EPS = 1E-12

            tailrec fun sin(x: Double, n: Int = 0, result: Double = 0.0): Double {
              val m = 2 * n + 1
              val z = (power(-1.0, n) * power(x, m)) / fakultaet(m)
              val result2 = result + z
              println("n: $n -> result=$result2 (z=$z)")

              if (abs(z) < EPS) {
                return result2;
              }

              return sin(x, n + 1, result2)
            }

            fun functionsTailRecursion() {
              println("sin(1.0) = ${sin(1.0)}")
            }

            fun main() = functionsTailRecursion()
* Die rekursive Funktion, wird durch den Compiler, in eine Variante mit einer schnellen und effizienten schleifenbasierte Version optimiert
* Damit wird das Risiko eines Stapelüberlaufs (Stack overflow) umgangen
* Der rekursive Aufruf muss die letzte Operation sein

Inline Funktionen

Warum Inline Funkionen?

  • Lambdas haben einen Overhead
    • Lambdas sind Objekte
    • Variablen und Argumente müssen gespeichert
    • …
  • Durch Inlining wird der Bytecode direkt an die Aufrufstelle gesetzt
  • Erhöht den Anteil des generierten Bytecodes
* Inlining liefert Leistungsverbesserungen
* Manche Lambdas werden hier exklusiv verwendet

Inline Funkionen

In Kotlin …


            inline fun inlined(lambda: () -> Unit) {
              println("Starting ...")
              lambda()
              println("Finished!")
            }

            fun main() {
              inlined { println("Hello, World!") }
            }
Ähnlich wie die Template-Klassen des Spring Frameworks

Inline Funkionen

Datei: FunctionsInline.kt

            package de.rahn.kotlin.lang

            import java.util.concurrent.locks.Lock
            import java.util.concurrent.locks.ReentrantLock

            inline fun <T> lockTemplate(lock: Lock, body: () -> T): T {
              lock.lock()
              try {
                return body()
              } finally {
                lock.unlock()
              }
            }

            fun functionsInline() {
              val s: String = lockTemplate(ReentrantLock()) {
                "Synchronized call"
              }
              println(s)
            }

            fun main() = functionsInline()
* Reduziert den Overhead von Lambdas
* Der Inhalt der Funktion und das übergebene Lambda, wird durch den Compiler, direkt an die Aufrufstelle des Aufrufs eingesetzt
* Interessant bei Funktionen, die eine Fehlerbehandlung durchführen und eine Ausnahme werfen. Die Ausnahme wird dann an der Stelle geworfen, wo der Fehler ausgetreten ist!
