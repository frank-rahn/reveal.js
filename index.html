Explicit-return

Datei: FunctionsExplicitReturn.kt

            package de.rahn.kotlin.lang

            import kotlin.math.abs

            fun sin(x: Double): Double {
              var result = 0.0

              for (n in 0..100) {
                val z = (power(-1.0, n) * power(x, 2 * n + 1)) / fakultaet(2 * n + 1)
                result += z
                println("n: $n -> result=$result (z=$z)")

                if (abs(z) < 0.000_000_000_001) {
                  break
                }
              }

              return result
            }

            fun functionsExplicitReturn() {
              println("sin(1.0) = ${sin(1.0)}")
            }

            fun main() = functionsExplicitReturn()
  • Dieser Algorithmus ist eine Näherung für sinus, die mit |z| < 10^-12 abbricht (genau genug)
  • Die Implementierung des Sinus wird auf den nächsten Seiten wiederverwendet

Tail-recursive (Verwendet Tail-recursion/call oder Endrekursion

Datei: FunctionsTailRecursion.kt

            package de.rahn.kotlin.lang

            import kotlin.math.abs

            const val EPS = 1E-12

            tailrec fun sin(x: Double, n: Int = 0, result: Double = 0.0): Double {
              val m = 2 * n + 1
              val z = (power(-1.0, n) * power(x, m)) / fakultaet(m)
              val result2 = result + z
              println("n: $n -> result=$result2 (z=$z)")

              if (abs(z) < EPS) {
                return result2;
              }

              return sin(x, n + 1, result2)
            }

            fun functionsTailRecursion() {
              println("sin(1.0) = ${sin(1.0)}")
            }

            fun main() = functionsTailRecursion()
  • Die rekursive Funktion, wird durch den Compiler, in eine Variante mit einer schnellen und effizienten schleifenbasierte Version optimiert
  • Damit wird das Risiko eines Stapelüberlaufs (Stack overflow) umgangen
  • Der rekursive Aufruf muss die letzte Operation sein

Inline Funktionen

Datei: FunctionsInline.kt

            package de.rahn.kotlin.lang

            import java.util.concurrent.locks.Lock
            import java.util.concurrent.locks.ReentrantLock

            inline fun <T> lockTemplate(lock: Lock, body: () -> T): T {
              lock.lock()
              try {
                return body()
              } finally {
                lock.unlock()
              }
            }

            fun functionsInline() {
              val s: String = lockTemplate(ReentrantLock()) {
                "Synchronized call"
              }
              println(s)
            }

            fun main() = functionsInline()
  • Reduziert den Overhead von Lambdas
  • Der Inhalt der Funktion und das übergebene Lambda, wird durch den Compiler, direkt an die Stelle des Aufrufs eingesetzt
  • Interessant bei Funktionen, die eine Fehlerbehandlung durchführen und eine Ausnahme werfen. Die Ausnahme wird dann an der Stelle geworfen, wo der Fehler ausgetreten ist!
Datei: FunctionsInlineNoinline.kt

            package de.rahn.kotlin.lang

            import java.util.concurrent.locks.Lock
            import java.util.concurrent.locks.ReentrantLock

            inline fun <T> lockTemplate2(noinline lock: () -> Lock, body: () -> T): T {
              val lockObj = lock()
              lockObj.lock()
              try {
                return body()
              } finally {
                lockObj.unlock()
              }
            }

            fun functionsInline() {
              val s: String = lockTemplate2({ ReentrantLock() }) {
                "Synchronized call"
              }
              println(s)
            }

            fun main() = functionsInline()

            package de.rahn.kotlin.lang.classes

            class Person(name: String) {
              init {
                println("Init block No. 1: $name")
              }

              val lastName = name

              init {
                println("Init block No. 2")
              }

              var age = -1

              init {

              }

                this.age = age
              }

              init {

              }

  • Wenn der Block Nr. 1 ausgeführt wird, gibt es noch keine Properties
  • In den Blöcken können die Argumente des primären Konstruktor genutzt werden
  • Wenn der Block Nr. 2 ausgeführt wird, steht der Nachnamen zu Verfügung
  • Die Reihenfolge muss berücksichtigt werden
  • Kann eingesetzt werden um die Implementierung zu strukturieren
  • Sollte sparsam eingesetzt werden

Vererbung von Interfaces und
Auflösen von Override Konflikten


            package de.rahn.kotlin.lang.classes

            interface Info {
                fun infos(): Array<String>
            }

            interface Essen : Info {
                fun essen()

                override fun infos(): Array<String> {
                    return arrayOf("Muss essen")
                }
            }

            interface Sterben : Info {
                fun sterben() {
                    println("Röchel... Tot")
                }

                override fun infos(): Array<String> {
                    return arrayOf("Kann sterben")
                }
            }

            abstract class Chordatier(val art: String = "Chordata")
                : Sterben, Essen {
                val hatHerz: Boolean
                    get() = true

                override fun infos(): Array<String> {
                    return super<Sterben>.infos() +
                            super<Essen>.infos() +
                            "Hat ein Herz"
                }
            }

            open class Wirbeltier(art: String = "Vertebrata")
                : Chordatier(art) {
                val hatWirbelsaeule: Boolean
                    get() = true

                override fun essen() {
                    TODO("Not yet implemented")
                }

                override fun infos(): Array<String> {
                    return super.infos() +
                            "Hat eine Wirbelsäule"
                }
            }

            open class Saeugetier(art: String = "Mammalia")
                : Wirbeltier(art) {
                val saeugtNachwuchs: Boolean
                    get() = true

                override fun infos(): Array<String> {
                    return super.infos() +
                            "Säugt sein Nachwuchs"
                }
            }
  • Vererbung von Interfaces, ähnlich wie in Java
  • Vorgriff: Collections / Arrays
  • Gibt es einen Konflikt, so muss die Klasse eine Member Funktion bereitstellen und dort den Konflikt auflösen
    • Es hilft nicht die Klasse abstrakt zu machen
  • Verwendung von super
  • Vorgriff: Der + ist ein Infix Operator
    • Wegen dem Infix muss das + am Zeilenende stehen (Stichwort)
